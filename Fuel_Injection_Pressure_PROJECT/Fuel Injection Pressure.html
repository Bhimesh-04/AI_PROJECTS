In [54]:
BTE_model = RandomForestRegressor(n_estimators=100,random_state=42)
BTE_model.fit(X_train,y1_train)

y1_predict = BTE_model.predict(X_test)

print("Mean_absolute_error is : ",mean_absolute_error(y1_test,y1_predict))
print("Mean_squared_error is : ",mean_squared_error(y1_test,y1_predict))
print("Root_Mean_squared_error is : ",np.sqrt(mean_squared_error(y1_test,y1_predict)))
print("R2_score is : ",r2_score(y1_test,y1_predict))

plt.plot(y1_test, y1_test, 'r.', label="Actual BTE")  # Red dots for actual values
plt.plot(y1_test, y1_predict, 'b-', label="Predicted BTE")  # Blue line for predictions
plt.xlabel("Actual BTE (%)")
plt.ylabel("Predicted BTE (%)")
plt.title("BTE MODEL - Actual vs Predicted")
plt.legend()
plt.show()
